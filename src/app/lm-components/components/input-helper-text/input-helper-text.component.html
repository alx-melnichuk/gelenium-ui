Helper text

The height of the input element depends on the presence of the "helperText" auxiliary text. If two input elements are located on the same line and the first has auxiliary text, and the second does not, then they will differ in height. It will look like this:

{{ labelShowSource }} 
<form [formGroup]="formGroup06" novalidate>

  <div style="display: flex; align-items: center;">
    <grn-input class="in-item" formControlName="input06a"
      exterOutlined label="Enter the name"
      helperText="The name must be complete.">
    </grn-input>

    <grn-input class="in-item" formControlName="input06b"
      exterOutlined label="Enter the name">
    </grn-input>
  </div>

</form>
 
import { Component, ViewEncapsulation } from '@angular/core';
import { FormControl, FormGroup } from '@angular/forms';

@Component({
  selector: 'app-input',
  templateUrl: './input.component.html',
  styleUrls: ['./input.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class InputComponent {
  public formGroup06: FormGroup = new FormGroup({
    input06a: new FormControl('Hello World', []),
    input06b: new FormControl('Hello World', []),
  });
}
 
.in-item {
  width: 240px;
  margin: 10px 10px;
}

For a quick fix, you can add a helperText property with a space value to the second input element (helperText=" ").
