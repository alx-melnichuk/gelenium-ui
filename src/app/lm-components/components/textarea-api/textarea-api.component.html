API: GrnTextarea

Selector: grn-textarea

Exported as: grnTextarea

Properties

Name Description
@Input() id: string; ID of the native textarea element.
(default = 'grn_textarea_' + ++identifier;)
@Input() label: string; The content of the label.
(default = '';)
@Input() exterior: string | null; Determines the appearance of an element.
It can take the following values:
  • "outlined";
  • "underline";
  • "standard" (default);
@Input() isReadOnly: string | null; Whether the element is readonly.
Attribute (no content).
@Input() isRequired: string | null; Adds a validation condition: required field.
Attribute (no content).
@Input() isDisabled: string | null; The element is disabled.
Attribute (no content).
@Input() lbShrink: string | null; Disable label animation. The label is always static, at the top.
Attribute (no content).
@Input() frameSize: string | null; Determines the size of the element.
It can take the following values:
  • "short" - if one line is height: 38px;
  • "small" - if one line is height: 44px;
  • "middle" - if one line is height: 50px;
  • "normal" - if one line is height: 56px;
  • "large" - if one line is height: 62px;
  • "huge" - if one line is height: 68px;
@Input() hiddenLabel: string | null; Disables the display of the label.
Attribute (no content).
@Input() isError: string | null; Forces the error status of an item to be displayed.
Attribute (no content).
@Input() autoComplete: string; Defines permissions for autocomplete native textarea.
(default = '';)
@Input() helperText: string; Specifies the text for additional clarification or error.
(default = null;)
@Input() wdFull: string; Sets the parent's width (width: 100%;).
Attribute (no content).
@Input() minLength: number; Defines the minimum number of characters for an textarea element.
(default = null;)
@Input() maxLength: number; Defines the maximum number of characters for an textarea element.
(default = null;)
@Input() minRows: number; Defines the minimum number of visible rows. The minimum height of the text area is calculated as "minRows" multiplied by to the line height. Active if "cntRows" is not specified.
(default = null;)
@Input() cntRows: number; Defines the number of visible rows. The height of the text area is calculated as "cntRows" multiplied by to the line height.
(default = null;)
@Input() maxRows: number; Defines the maximum number of visible rows. The maximum height of the text area is calculated as "maxRows" multiplied by to the line height. Active if "cntRows" is not specified.
(default = null;)
@Input() cntCols: number; Determines the number of columns.
(default = null;)
@Output() inputData: EventEmitter<Event>; The event is called when the user changes the value of an element.
@Output() changeData: EventEmitter<Event>; The event is called when the user has finished changing the value of an element.
@Output() keydownData: EventEmitter<KeyboardEvent>; The event is called when any button moves down.
@Output() keypressData: EventEmitter<KeyboardEvent>; The event is called when a button that has a printable character is pressed.
@Output() keyupData: EventEmitter<KeyboardEvent>; The event is called when any button moves up.
