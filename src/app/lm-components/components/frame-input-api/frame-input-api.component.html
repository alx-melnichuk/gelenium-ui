API: GrnFrameInput

Selector: grn-frame-input

Exported as: grnFrameInput

Properties

Name Description
@Input() inputId: string; Data entry item identifier. Specified in the value of the "for" attribute for the label.
(default = '';)
@Input() label: string; The content of the label.
(default = '';)
@Input() exterior: string | null; Determines the appearance of an element.
It can take the following values:
  • "outlined";
  • "underline";
  • "standard" (default);
@Input() isRequired: boolean; Adds a validation condition: required field.
(default = false;)
@Input() isDisabled: boolean; The element is disabled.
(default = false;)
@Input() isFocused: boolean; Indicates that an input element has received input focus.
(default = false;)
@Input() isFilled: boolean; Indicates that the input element is not empty (filled).
(default = false;)
@Input() isLabelShrink: boolean; An indication that the label has moved up and shrunk.
(default = false;)
@Input() hiddenLabel: boolean; Disables the display of the label.
(default = false;)
@Input() isError: boolean; Forces the error status of an item to be displayed.
(default = false;)
@Input() helperText: string; Specifies the text for additional clarification or error.
(default = null;)
@Input() frameSize: FrameSize; Determines the size of the element.
It can take the following values:
  • "short" - if one line is height: 38px;
  • "small" - if one line is height: 44px;
  • "middle" - if one line is height: 50px;
  • "normal" - if one line is height: 56px;
  • "large" - if one line is height: 62px;
  • "huge" - if one line is height: 68px;
