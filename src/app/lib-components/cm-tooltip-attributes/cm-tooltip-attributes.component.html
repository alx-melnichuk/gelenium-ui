The following attributes are supported:
  • "config" - Defines a configuration object. More details.
  • "glnTooltip" - specifies the tooltip text;
  • "glnttPosition" - specifies the position of the tooltip;
    This property can take the following values:
    • 'bottom' (default);
    • 'bottom-start';
    • 'bottom-end';
    • 'top';
    • 'top-start';
    • 'top-end';
    • 'left';
    • 'left-start';
    • 'left-end';
    • 'right';
    • 'right-start';
    • 'right-end';
Attribute: glnttPosition.

@Component({
  selector: 'app-tooltip',
  templateUrl: './tooltip.component.html',
  styleUrls: ['./tooltip.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class TooltipComponent {
  public showDelay02e: number = 500;
  public hideDelay02e: number = 500;

  public incVal(value: number, delta: number, max: number): number {
    return value + delta <= max
      ? value + delta : value > max ? max : value;
  }
  public decVal(value: number, delta: number, min: number): number {
    return value - delta >= min
      ? value - delta : value < min ? min : value;
  }

}

    'right-start',
    'right-end',
    'left',
    'left-start',
    'left-end',

  public getValue(eventTarget: any): any {
    return eventTarget != null ? (eventTarget as Attr).value : '';
  }
}
