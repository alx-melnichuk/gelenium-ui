Ornaments. #

For the GlnInput component, it is possible to display an ornament (picture or text).
To display the ornament on the left, you need to add the "gln-orn-lf" attribute to the element inside the "gln-input" content.
To display the ornament on the right, you need to add the "gln-orn-rg" attribute to the element inside the "gln-input" content.

Example: 
<gln-input formControlName="model01a"
  label="Enter value">
  <div gln-orn-lf><span>left</span></div>
  <div gln-orn-rg><span>right</span></div>
</gln-input>

Adding a prefix or suffix does not disable legend animation for that element.
For blocks: <span>left</span> and <span>right</span> you need to specify the value of horizontal indents.

You can increase/decrease the font size for an input element. For example, when entering a password, you can increase the font size to display larger black circles.
To do this, add a style: 

.io-ornament .io-passwd-big .glnin-input {
  font-size: 20px;
}

In this case, you need to add the "wdFull" attribute, otherwise when you change the font size, the length of the input element will change.

You can add the "isLabelShrink" attribute and the legend will be fixed in the title. Nor will the legend descend to the pre-holder position.

import { Component, ViewEncapsulation } from '@angular/core';
import { FormControl, FormGroup } from '@angular/forms';

@Component({
  selector: 'app-input',
  templateUrl: './input.component.html',
  styleUrls: ['./input.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class InputComponent {
  public minLength05 = 3;
  public maxLength05 = 15;
  public controls05 = {
    model05a: new FormControl('', []),
    model05b: new FormControl('thirty', []),
    model05c: new FormControl('one', [
      Validators.required,
      Validators.minLength(this.minLength05),
      Validators.maxLength(this.maxLength05)
    ]),
    model05d: new FormControl('', [
      Validators.required,
      Validators.minLength(this.minLength05),
      Validators.maxLength(this.maxLength05),
    ]),
    model05e: new FormControl('thirty', [Validators.required]),
    model05f: new FormControl('thirty', [Validators.required]),
    model05g: new FormControl('thirty', [Validators.required]),
    model05h: new FormControl('thirty', [Validators.required]),
    model05i: new FormControl('', [Validators.required]),
    model05j: new FormControl('test', []),
    model05k: new FormControl('test', []),
  };
  public formGroup05: FormGroup = new FormGroup(this.controls05);
  public exterior05 = 'outlined';
  public isBtnEyeCrossed05e = false;
  public isBtnEyeCrossed05f = false;
  public isBtnEyeCrossed05g = false;
  public isBtnEyeCrossed05h = false;
  // GlnInputConfig
  public config05h = {
    ornamRgAlign: 'baseline',
  };
  public ornamLfAlign05 = 'baseline';
  public ornamRgAlign05 = 'baseline';
  // GlnInputConfig
  public config05 = {
    ornamLfAlign: this.ornamLfAlign05,
    ornamRgAlign: this.ornamRgAlign05,
  };

  constructor() {}

  public convert(ornamAlign: string | null): OrnamAlign {
    return OrnamAlignUtil.convert(ornamAlign) || OrnamAlign.default;
  }
  public changeConfig05(
    ornamLfAlign: string | undefined,
    ornamRgAlign: string | undefined
  ): void {
    this.config05 = {
      ornamLfAlign: GlnFrameOrnamAlignUtil.convert(ornamLfAlign || null)?.toString() || 'default',
      ornamRgAlign: GlnFrameOrnamAlignUtil.convert(ornamRgAlign || null)?.toString() || 'default',
    };
  }
  public toStr(value: number): string {
    return String(value);
  }
  public toNum(value: any): number {
    return Number(value);
  }
}
