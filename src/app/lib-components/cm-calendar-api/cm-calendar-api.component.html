dateClasses1 = (
  date: Date, view: string, currentDate: Date
): string[] => {
  const result: string[] = [];
  const day: number = date.getDate();
  if ('day' === view
    && date.getMonth() === currentDate.getMonth()
    && (day === 1 || day === 20)) {
    result.push('cnat6-demo1');
  }

};
