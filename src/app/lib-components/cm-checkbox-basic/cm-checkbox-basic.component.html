Basic. #

To use this component, in your module you need to add: 
import { GlnCheckboxModule } from 'gelenium-ui';

The GlnCheckbox component displays the behavior of a simple "state checkbox" property (<input type="checkbox">). A component allows you to select one or more of these components.

For example: 
<gln-checkbox formControlName="model01">
</gln-checkbox>
Checked.
Unchecked.
Indeterminate.
Disabled Checked.
Disabled Unchecked.
Disabled Indeterminate.

import {
  Component, ViewEncapsulation
} from '@angular/core';
import {
  FormControl, FormGroup
} from '@angular/forms';

@Component({
  selector: 'app-checkbox',
  templateUrl: './checkbox.component.html',
  styleUrls: ['./checkbox.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class CheckboxComponent {
  public formGroup01: FormGroup = new FormGroup({
    model01a: new FormControl(true, []),
    model01b: new FormControl(false, []),
    model01c: new FormControl(true, []),
    model01d: new FormControl(true, []),
    model01e: new FormControl(false, []),
    model01f: new FormControl(false, []),
  });
}
