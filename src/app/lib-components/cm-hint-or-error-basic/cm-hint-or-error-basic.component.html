Basic. #

To use this component, in your module you need to add: 
import { GlnHintOrErrorModule } from 'gelenium-ui';

The GlnHintOrErrorComponent component is a simple utility component. Its main tasks are:

  • display hint text; (color: var(--glnhe-def-cl));
  • display error text (color: var(--glnhe-err-cl));
  • display hint text when the input element has received focus (color: var(--glnhe-foc-cl));
  • display error when the input element has received focus (color: var(--glnhe-foc-err-cl));
  • display hint on disabled (color: var(--glnhe-dis-cl));

All css variables are global css variables from the GlnFrame component. This component is used in GlnInputComponent and GlnTextareaComponent.
Here is an example of usage: 

  <gln-hint-or-error *ngIf="!!helperText"
    [text]="helperText"
    [isError]="!!formControl.errors || isError2"
    [isFocused]="isFocused"
    [isDisabled]="isDisabled2">
  </gln-hint-or-error>

import {
  Component, ViewEncapsulation
} from '@angular/core';
import {
  FormControl, FormGroup
} from '@angular/forms';

@Component({
  selector: 'app-hint-or-error',
  templateUrl: './hint-or-error.component.html',
  styleUrls: ['./hint-or-error.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class HintOrErrorComponent {
  public controls01 = {
    model01a: new FormControl('is', []),
    model01b: new FormControl('demo', []),
  };
  public formGroup01: FormGroup = new FormGroup(this.controls01);
  public isDisabled01 = false;
  public isFocusInput01a = false;
  public isFocusInput01b = false;

  public changeIsDisabled(isDisabled01: boolean): boolean): void {
    if (isDisabled01) {
      this.formGroup01.disable();
    } else {
      this.formGroup01.enable();
    }
    return isDisabled01;
  }

}
