Size #

The "cell-size" property determines the cell size of the element (its height and width).

It takes the following values:

  • "short" - sets the height to 28px;
  • "small" - sets the height to 32px;
  • "middle" - sets the height to 36px (default);
  • "wide" - sets the height to 40px;
  • "large" - sets the height to 44px;
  • "huge" - sets the height to 48px;
"Short"height:28px;font-size:14px
"Small" height: 32px; font-size: 15px;
"Middle" height: 36px; font-size: 16px;
"Wide" height: 40px; font-size: 16px;
"Large" height: 44px; font-size: 18px;
"Huge" height: 48px; font-size: 20px;

import {
  Component, ViewEncapsulation
} from '@angular/core';

@Component({
  selector: 'app-calendar',
  templateUrl: './calendar.component.html',
  styleUrls: ['./calendar.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class CalendarComponent {
  public selectedDate: Date = this.getSelectedDate();

  public value03a: Date | null = new Date(this.selectedDate);
  public value03b: Date | null = new Date(this.selectedDate);
  public value03c: Date | null = new Date(this.selectedDate);
  public value03d: Date | null = new Date(this.selectedDate);
  public value03e: Date | null = new Date(this.selectedDate);
  public value03f: Date | null = new Date(this.selectedDate);

  private getSelectedDate(): Date {
    const d1: Date = new Date();
    const result: Date = new Date(d1.getFullYear(), d1.getMonth(), d1.getDate(), 0, 0, 0, 0);
    result.setDate(result.getDate() + (d1.getDate() === 1 ? 1 : -1));
    return result;
  }
}
