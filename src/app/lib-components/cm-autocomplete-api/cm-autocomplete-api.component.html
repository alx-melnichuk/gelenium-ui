--glnac-border-radius:
 (triggerRect.height / 10) + 'px'; 		Defines the value of the "border-radius" property for the options list panel. This will make rounded edges for the options dropdown panel.
(default "(triggerRect.height / 10) + 'px'"). Example: 
<gln-input formControlName="model01a"
  [glnAutocompleteTrigger]="autocomplete01">
</gln-input>
<gln-autocomplete #autocomplete01="glnAutocomplete"
  class="border-radius">
  ...
<gln-autocomplete>
 
gln-autocomplete.border-radius {
--glnac-border-radius: 7px;
}
--glnac-elevation:
