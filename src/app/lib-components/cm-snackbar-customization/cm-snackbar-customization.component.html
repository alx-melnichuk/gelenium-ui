Customization #

You can specify your own context when displaying the snackbar.

For example: 
<ng-template #contextTemplate1 let-context="context" let-snackbarRef="snackbarRef">
  <div class="sccs-alert">
    <span>
      <b>{{context.name}}</b> is a very <em>tasty</em>&nbsp;<u>fruit</u> !
    </span>
    <div>
      <button class="sccs-close" (click)="snackbarRef.close('close')">X</button>
    </div>
  </div>
</ng-template>

<button class="sncs-btn"
  tabindex="0"
  type="button"
  (click)="clickDemo(contextTemplate1, { name: 'Mango' })">
  <span> Action </span>
</button>
 
import { Component, ViewEncapsulation } from '@angular/core';
import { GlnSnackbarService } from 'gelenium-ui';

@Component({
  selector: 'app-snackbar',
  templateUrl: './snackbar.component.html',
  styleUrls: ['./snackbar.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class SnackbarComponent {

  constructor(private snackbarService: GlnSnackbarService) {}

  public clickDemo(template: TemplateRef<any>, context: Record<string, any>): void {
    this.snackbarService.openFromTemplate(template, { data: context })
    .result.then((response) => {
      console.log(`Template response=${response}`);
    });
  }
}

import {
  Component, ViewEncapsulation
} from '@angular/core';
import { GlnSnackbarService } from 'gelenium-ui';

@Component({
  selector: 'app-snackbar',
  templateUrl: './snackbar.component.html',
  styleUrls: ['./snackbar.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class SnackbarComponent {

  constructor(private snackbarService: GlnSnackbarService) {}

  public clickTemplate(
    template: TemplateRef<any>, context: Record<string, unknown> | null
  ): void {
    this.snackbarService.openFromTemplate(template, { data: context })
    .result.then((response) => {
      console.log(`Template response=${response}`);
    });
  }

}
