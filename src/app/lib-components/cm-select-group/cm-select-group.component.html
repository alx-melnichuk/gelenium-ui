Group #

It is possible to group some options in the selection list. For this, the "GlnOptionGroup" component is used.

For example: 
<gln-select formControlName="model01a"
  label="Enter value"
  wdFull>
  <gln-option><em>None</em></gln-option>
  <gln-option-group label="fruits">
    <gln-option value="mango">Mango</gln-option>
    <gln-option value="lemon">Lemon</gln-option>
    <gln-option value="orange">Orange</gln-option>
    <gln-option value="kiwi">Kiwi</gln-option>
  </gln-option-group>
</gln-select>

Attributes are also supported:

  • "label" - the name of the option group.
  • "isDisabled" - a sign that this group of options is disabled (isDisabled);
exterior: {{ labelOutlined }} {{ labelUnderline }} {{ labelStandard }}

import {
  Component, ViewEncapsulation
} from '@angular/core';
import {
  FormGroup, FormControl 
} from '@angular/forms';

@Component({
  selector: 'app-select',
  templateUrl: './select.component.html',
  styleUrls: ['./select.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class SelectComponent {

  public exterior04a = 'outlined';

  public tubers = ['potato', 'jerusalem artichoke;', 'sweet potato', 'cassava', 'tuberous nasturtium'];
  public cabbage = ['white cabbage', 'portuguese cabbage', 'japanese cabbage', 'kohlrabi', 'broccoli'];
  public salad = ['leaf parsley', 'mustard leaf', 'spinach', 'chard (leaf beet)', 'vegetable fennel'];

  public control04a = {
    model04a: new FormControl(this.tubers[1], []),
  };
  public formGroup04a: FormGroup = new FormGroup(this.control04a);

  public capitalizeFirstLetter(value: string): string {
    return value ? value[0].toUpperCase() + value.slice(1) : '';
  }
}
