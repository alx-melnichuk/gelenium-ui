API: GlnFrame

Selector: gln-frame

Exported as: GlnFrame

Properties

Name Description
@Input() config: GlnFrameConfig | null = null; Defines a configuration object.
@Input() exterior: Exterior | null = null; Determines the appearance of an element.
It can take the following values:
  • "outlined";
  • "underline";
  • "standard";
@Input() isDisabled: boolean | null = null; The element is disabled.
(default = false;)
@Input() isError: boolean | null = null; Forces the error status of an item to be displayed.
(default = false;)
@Input() isFilled: boolean | null = null; Indicates that the input element is not empty (filled).
(default = false;)
@Input() isHoverColor: boolean | null = null; If specified, then when hovering the mouse, the color of the label and frame will be the same as when the focus was received.
If not specified, then when you hover the mouse, the color of the label and frame will be the same as the default.
@Input() isLabelShrink: string | boolean | null | undefined; An indication that the label has moved up and shrunk.
@Input() isNoAnimation: boolean | null = null; Disable animation execution.
@Input() isNoLabel: boolean | null = null; Hide label display.
@Input() isRequired: boolean | null = null; Adds a validation condition: required field.
(default = false;)
@Input() label: string
  | null
  | undefined; 		The content of the label.
