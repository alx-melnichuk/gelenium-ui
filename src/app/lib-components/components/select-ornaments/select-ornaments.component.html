Ornaments. #

For the GlnSelect component, it is possible to display an ornament (picture or text).
To display the ornament on the left, you need to add the "gln-orn-lf" attribute to the element inside the "gln-select" content.
To display the ornament on the right, you need to add the "gln-orn-rg" attribute to the element inside the "gln-select" content.

Example: 
<gln-select formControlName="model01a"
  label="Enter value">
  <div gln-orn-lf><span>left</span></div>
  <div gln-orn-rg><span>right</span></div>
  <gln-option><em>None</em></gln-option>
  <gln-option value="mango">Mango</gln-option>
  <gln-option value="lemon">Lemon</gln-option>
  <gln-option value="orange">Orange</gln-option>
  <gln-option value="kiwi">Kiwi</gln-option>
</gln-select>

Adding a prefix or suffix does not disable legend animation for that element.
For blocks: <span>left</span> and <span>right</span> you need to specify the value of horizontal indents.

For example, let's build a display of Unicode characters.

exterior: {{ labelOutlined }} {{ labelUnderline }} {{ labelStandard }}
Section: "gln-orn-lf" + uncode.
{{ capitalizeFirstLetter(item) }}
Section: "gln-orn-rg" + uncode.
{{ capitalizeFirstLetter(item) }}
{{ labelShowSource }}

 ChangeDetectionStrategy, Component, Input, ViewEncapsulation
 } from '@angular/core';
import {
 FormControl, FormGroup
 } from '@angular/forms';
import {
 GlnFrameConfig, OrnamAlign
 } from 'gelenium-ui';

@Component({
  selector: 'app-input',
  templateUrl: './input.component.html',
  styleUrls: ['./input.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class SelectComponent {

  public exterior06a = 'outlined';

  public salad = [
    'lettuce (lettuce)',
    'leaf parsley',
    'mustard leaf',
    'swans',
    'sorrel',
    'spinach',
    'purslane',
    'chard (leaf beet)',
    'vegetable fennel',
    'arugula',
    'cucumber grass',
    'nettles',
  ];
  public isUncode06a = false;
  public isUncode06b = false;
  public control06a = {
    model06a: new FormControl([], []),
    model06b: new FormControl([], []),
  };
  public formGroup06a: FormGroup = new FormGroup(this.control06a);

  public capitalizeFirstLetter(value: string): string {
    return value ? value[0].toUpperCase() + value.slice(1) : '';
  }
}
