API: GrnFrameInput

Selector: grn-frame-input

Exported as: grnFrameInput

Properties

Name Description
@Input() label: string; The content of the label.
(default = '';)
@Input() config: GrnFrameInputConfig | null = null; Defines a configuration object.
@Input() exterior: Exterior | null = null; Determines the appearance of an element.
It can take the following values:
  • "outlined";
  • "underline";
  • "standard";
@Input() frameSize: FrameSize | null = null; Determines the size of the element.
It can take the following values:
  • "short" - if one line is height: 38px;
  • "small" - if one line is height: 44px;
  • "middle" - if one line is height: 50px (default);
  • "wide" - if one line is height: 56px;
  • "large" - if one line is height: 62px;
  • "huge" - if one line is height: 68px;
@Input() isLabelShrink: boolean | null = null; An indication that the label has moved up and shrunk.
@Input() hiddenLabel: boolean | null = null; Disables the display of the label.
@Input() isDisabled: boolean; The element is disabled.
(default = false;)
@Input() isFocused: boolean; Indicates that an input element has received input focus.
(default = false;)
@Input() isFilled: boolean; Indicates that the input element is not empty (filled).
(default = false;)
@Input() isError: boolean; Forces the error status of an item to be displayed.
(default = false;)
@Input() isRequired: boolean; Adds a validation condition: required field.
(default = false;)
@Output() readonly clickFrame: EventEmitter<Event> = new EventEmitter(); Defines a click event on the element's frame
