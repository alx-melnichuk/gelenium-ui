Border radius.

The GlnButton component has the ability to change the "border-radius" css property. This requires creating a css variable "--glnb-br-rd" at the level of the component or any parent element. The value of this css variable can be any valid (as described by "border-radius" ) value.

You can also change the value of the left / right indents, for a more convenient perception.

exterior: {{ labelContained }} {{ labelOutlined }} {{ labelText }}

border-radius = ( BaseSize * {{ borderRadius }}/100)

import {
  Component, ViewEncapsulation
} from '@angular/core';

@Component({
  selector: 'app-button',
  templateUrl: './button.component.html',
  styleUrls: ['./button.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class ButtonComponent {
  public exterior04a = 'outlined';
  public routerLink = '/components/button';
  public fragment04a = 'size-link4a';
  public borderRadius = 50;
  public propRadius = '--glnb-br-rd';
  public isAddPadding04a = true;
  public labelPdRatio = 0.4;

  public sizeShort = GlnFrameSize.short;

  public getLabel(exterior: string): string {
    return exterior ?
      exterior.substring(0, 1).toUpperCase() + exterior.substring(1) : '';
  }
  public getValue(value: GlnFrameSize): number {
    return GlnFrameSizeUtil.getValue(value) || 0;
  }
  public round(value: number): number {
    return Math.round(100 * value) / 100;
  }
  public getConfig(labelPd: number): GlnButtonConfig {
    return { labelPd };
  }
}

  Component, ViewEncapsulation
} from '@angular/core';

@Component({
  selector: 'app-button',
  templateUrl: './button.component.html',
  styleUrls: ['./button.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class ButtonComponent {
  public exterior04b = 'outlined';
  public routerLink = '/components/button';
  public fragment04k = 'size-link4k';
  public propRadius = '--glnb-br-rd';
  public isAddPadding04b = true;
  public radiusRatio = 0.3;
  public sizeMiddle = GlnFrameSize.middle;

  public getLabel(exterior: string): string {
    return exterior ?
      exterior.substring(0, 1).toUpperCase() + exterior.substring(1) : '';
  }
  public getValue(value: GlnFrameSize): number {
    return GlnFrameSizeUtil.getValue(value) || 0;
  }
  public round(value: number): number {
    return Math.round(100 * value) / 100;
  }
  public getConfig(labelPd: number): GlnButtonConfig {
    return { labelPd };
  }
}
