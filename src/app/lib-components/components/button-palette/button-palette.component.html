Palette.

You can customize the color palette for the buttons. To do this, let's define our own values for the required CSS variables.

All the CSS classes that define the colors of the button element are children of the "gln-button" CSS classes. Let's add a custom CSS class "bp-specific" to define our colors.

Button with "contained" exterior. #

An example of defining css variables for buttons with a "contained" appearance.

Primary Primary Disabled Link Link Disabled
{{ labelShowSource }} 
<div class="bp-items bp-specific">

  <gln-button exterior="contained">
    <span>Primary</span>
  </gln-button>

  <gln-button exterior="contained"
    isDisabled>
    <span>Primary Disabled</span>
  </gln-button>

  <gln-button exterior="contained">
    <a [routerLink]="[routerLink]"
      [fragment]="fragment1">
      Link
    </a>
  </gln-button>

  <gln-button exterior="contained"
    isDisabled>
    <a [routerLink]="[routerLink]"
      [fragment]="fragment1">
      Link Disabled
    </a>
  </gln-button>

</div>
 
import {
  Component, ViewEncapsulation
} from '@angular/core';

@Component({
  selector: 'app-button',
  templateUrl: './button.component.html',
  styleUrls: ['./button.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class ButtonComponent {
  public routerLink = '/components/button2';
  public fragment1 = 'palette1';
}
 
.bp-items {
  display: flex;
  flex-wrap: wrap;
}
.bp-items > :first-child {
  margin-left: 1%;
}
.bp-items > :not(first-child) {
  margin-left: 5%;
}

// Exterior: "contained"
.bp-specific .gln-button.glnbt-contained {
  --glnbt-c-ripple-cl: #686aec5d;
  --glnbt-c-def-cl: #00acc1;
  --glnbt-c-def-bg-cl: #b2ebf299;
  --glnbt-c-hov-bg-cl: #acebec4d;
  --glnbt-c-dis-cl: #76b0b7;
  --glnbt-c-dis-bg-cl: #b2ebf233;
}
