Basic

The GlnHintOrErrorComponent component is a simple utility component. Its main tasks are:

  • display hint text; (color: var(--glnf-def-cl));
  • display error text (color: var(--glnf-err-cl2));
  • display hint text when the input element has received focus (color: var(--glnf-hov-cl));
  • display error when the input element has received focus (color: var(--glnf-err-cl));
  • display tooltip on disabled (color: var(--glnf-dis-cl));

All css variables are global css variables from the GlnFrame component. This component is used in GlnInputComponent and GlnTextareaComponent.
Here is an example of usage: 

  <gln-hint-or-error *ngIf="!!helperText"
    [text]="helperText"
    [isError]="!!formControl.errors || isError2"
    [isFocused]="isFocused"
    [isDisabled]="isDisabled2">
  </gln-hint-or-error>

import {
  Component, ViewEncapsulation
} from '@angular/core';
import {
  FormControl, FormGroup
} from '@angular/forms';

@Component({
  selector: 'app-textarea',
  templateUrl: './textarea.component.html',
  styleUrls: ['./textarea.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class TextareaComponent {
  public controls01 = {
    input01a: new FormControl('is', []),
    input01b: new FormControl('demo', []),
  };
  public formGroup01: FormGroup = new FormGroup(this.controls01);
  public isDisabled01 = false;
  public isFocusInput01a = false;
  public isFocusInput01b = false;

  public changeIsDisabled(): void {
    if (this.isDisabled01) {
      this.formGroup01.disable();
    } else {
      this.formGroup01.enable();
    }
  }

}
