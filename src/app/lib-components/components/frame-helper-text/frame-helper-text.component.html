Helper text #

The height of the input element depends on the presence of the "helperText" auxiliary text. If two input elements are located on the same line and the first has auxiliary text, and the second does not, then they will differ in height. It will look like this:

<form [formGroup]="formGroup04"
  novalidate
  autocomplete="off">

  <div style="display: flex; align-items: center;">

    <gln-input class="app-elem"
      formControlName="model04a"
      label="Enter the first name"
      helperText="The first name must be complete."
      [exterior]="exterior04">
    </gln-input>

    <gln-input class="app-elem"
      formControlName="model04b"
      label="Enter the last name"
      [exterior]="exterior04">
    </gln-input>

  </div>

</form>
 
import {
  Component, ViewEncapsulation
} from '@angular/core';
import {
  FormControl, FormGroup
} from '@angular/forms';

@Component({
  selector: 'app-frame',
  templateUrl: './frame.component.html',
  styleUrls: ['./frame.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class FrameComponent {
  public formGroup04: FormGroup = new FormGroup({
    model04a: new FormControl('William', []),
    model04b: new FormControl('Anderson', []),
  });
  public exterior04 = 'outlined';
}
 
.app-elem {
  min-width: 26ch;
  margin: 10px 10px;
}

For a quick fix, you can add a helperText property with a space value to the second input element (helperText=" ").
