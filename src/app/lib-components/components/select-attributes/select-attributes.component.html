Attributes

The following attributes are supported: #

  • "id" - parameter for defining the unique identifier of the field (id="unique_identifier");
  • "exterior" - Determines the appearance of an element.
    It can take the following values:
    • "outlined";
    • "underline";
    • "standard";
  • "isRequired" - a sign that the field is required (isRequired);
  • "helperText" - the parameter is used to provide context for a given field or error message (helperText="A value must be entered.");
  • "label" - The content of the label.
  • "noLabel" - a sign that the field has no label at all (noLabel);
  • "isDisabled" - a sign that the field is disabled (isDisabled);
  • "wdFull" - an attribute sets the width to the width of the parent (width: 100%).

import {
  Component, ViewEncapsulation
} from '@angular/core';
import {
  FormControl, FormGroup, Validators
} from '@angular/forms';

@Component({
  selector: 'app-select',
  templateUrl: './select.component.html',
  styleUrls: ['./select.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class SelectComponent {
  public fruits = ['mango', 'lemon', 'orange', 'kiwi'];

  public exterior02a = 'outlined';
  public formGroup02a: FormGroup = new FormGroup({
    model02a: new FormControl(null, []),
    model02b: new FormControl(this.fruits[1], [Validators.required]),
    model02c: new FormControl(this.fruits[1], []),
    model02d: new FormControl(this.fruits[1], []),
  });

  public capitalizeFirstLetter(value: string): string {
    return value ? value[0].toUpperCase() + value.slice(1) : '';
  }
}
