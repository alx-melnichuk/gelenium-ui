The following attributes are supported:

The following attributes are supported:

  • "noUppercase" is an attribute that cancels the display of text in uppercase;
  • "isDisabled" - a sign that the field is disabled (isDisabled);
  • "wdFull" is an attribute sets the width to the width of the parent (width: 100%);
  • "noElevation" is an attribute to remove button elevation (only for "contained");

import {
  Component, ViewEncapsulation
} from '@angular/core';

@Component({
  selector: 'app-button',
  templateUrl: './button.component.html',
  styleUrls: ['./button.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class ButtonComponent {
  public exterior02 = 'contained';
  public routerLink = '/components/button';
  public fragment = 'attribute-button';
  public getLabel(exterior: string): string {
    return exterior ?
      exterior.substring(0, 1).toUpperCase() + exterior.substring(1) : '';
  }
}
