Basic

The GrnButton component is a styled button.
It appears in three versions:

  • as text (exterior="text" - default)
  • as a container (exterior="contained")
  • as an outline (exterior="outlined")
Contained Outlined Text
{{ labelShowSource }} 
<div class="b-items">

  <grn-button exterior="contained"
    uppercase>
    Contained
  </grn-button>

  <grn-button exterior="outlined"
    uppercase>
    Outlined
  </grn-button>

  <grn-button exterior="text"
    uppercase>
    Text
  </grn-button>

</div>
 
import {
  Component, ViewEncapsulation
} from '@angular/core';
import {
  FormControl, FormGroup, Validators
} from '@angular/forms';

  templateUrl: './button.component.html',
  styleUrls: ['./button.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class ButtonComponent {
}
 
.b-items {
  display: flex;
  flex-wrap: wrap;
  &>* {
    margin-left: 10%;
  }
}
