Border radius. #

The GlnButton component has the ability to change the "border-radius" css property. This requires creating a css variable "--glnbt-br-rd" at the level of the component or any parent element. The value of this css variable can be any valid (as described by "border-radius" ) value.

You can also change the value of the left / right indents, for a more convenient perception.

exterior: {{ labelContained }} {{ labelOutlined }} {{ labelText }}

border-radius = ( BaseSize * {{ borderRadius }}/100)

import {
  Component, ViewEncapsulation
} from '@angular/core';
import {
  GlnFrameSizeUtil
} from 'gelenium-ui';

@Component({
  selector: 'app-button',
  templateUrl: './button.component.html',
  styleUrls: ['./button.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class ButtonComponent {
  public routerLink = '/components/button1';
  public exterior04a = 'outlined';
  public borderRadius = 50;
  public isAddPadding04a = true;
  public labelPdRatio = 0.4;

  public getLabel(exterior: string): string {
    return exterior ?
      exterior.substring(0, 1).toUpperCase() + exterior.substring(1) : '';
  }
  public getSize(frameSize: string | null): number {
    return GlnFrameSizeUtil.getValue(
      GlnFrameSizeUtil.convert(frameSize)) ?? 0;
  }
  public getRadius(frameSize: string | null, radiusRatio: number): number {
    return Math.round(
      (this.getSize(frameSize) * radiusRatio * 100) / 100);
  }
}
