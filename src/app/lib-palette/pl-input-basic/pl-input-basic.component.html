GlnInput is based on the GlnFrame component, which has its own palette by default.

To use the glnColor directive, you need to add in your module: 
import { GlnColorModule } from 'gelenium-ui';

import {
  Component, ViewEncapsulation
} from '@angular/core';
import {
  FormControl, FormGroup
} from '@angular/forms';

@Component({
  selector: 'app-input',
  templateUrl: './input.component.html',
  styleUrls: ['./input.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class InputComponent {
  public exterior01a = 'outlined';
  public control01a = {
    model01a: new FormControl(null, []),
    model01b: new FormControl(null, []),
    model01c: new FormControl('Hello World', []),
    model01d: new FormControl('Hello World', []),
  };
  public formGroup01a: FormGroup = new FormGroup(this.control01a);
}
