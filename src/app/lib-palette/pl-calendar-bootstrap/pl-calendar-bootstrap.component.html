It is possible to display GlnButton in style as "Bootstrap".
To do this, you need to specify the following configuration:

Bootstrap style

<div class="app-group">
  <div class="app-elem">
    <div>
      <small><em>Bootstrap style</em></small>
    </div>
    <div class="cnbs-clnd">
      <gln-calendar class="cnbs-bstr"
        [config]="cfgBs"
        [value]="value07h"
        [isDisabled]="isDisabled07h"
        (selected)="value07h=$event">
      </gln-calendar>
    </div>
  </div>
</div>
 
import {
  Component, ViewEncapsulation
} from '@angular/core';

@Component({
  selector: 'app-calendar',
  templateUrl: './calendar.component.html',
  styleUrls: ['./calendar.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class CalendarComponent {
  public isDisabled07h: boolean = false;
  public cfgBs = {
    cellSize: 30,
    isHideDayoff: true,
    isStartSunday: true,
    sizeDayWeek: 2
  };
  public selectedDate: Date = this.getSelectedDate();
  public value07h: Date | null = new Date(this.selectedDate);

  private getSelectedDate(): Date {
    const d1: Date = new Date();
    const result: Date = new Date(d1.getFullYear(), d1.getMonth(), d1.getDate(), 0, 0, 0, 0);
    result.setDate(result.getDate() + (d1.getDate() === 1 ? 1 : -1));
    return result;
  }
}
