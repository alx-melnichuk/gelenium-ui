Like a Material-UI. #

It is possible to display GlnCalendar in style as "MaterialUI".
To do this, you need to specify the following configuration:

 
Material-UI style
{{ labelShowSource }} 
<div class="app-group">
  <div class="app-elem">
    <div>
      <small><em>Material-UI style</em></small>
    </div>
    <div class="cnmt-clnd">
      <gln-calendar class="cnmt-mui"
        [value]="value07l"
        [config]="cfgMui"
        [isDisabled]="isDisabled07l"
        [isHideOldDays]="isHideOldDays07l"
        (change)="value07l=$event">
      </gln-calendar>
    </div>
  </div>
</div>
 
import {
  Component, ViewEncapsulation
} from '@angular/core';

@Component({
  selector: 'app-calendar',
  templateUrl: './calendar.component.html',
  styleUrls: ['./calendar.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class CalendarComponent {
  public isDisabled07l: boolean = false;
  public isHideOldDays07l: boolean = false;
  public cfgMui = {
    cellSize: 36,
    isHideDayoff: true,
    isHideOldDays: true,
    isStartSunday: true,
    sizeDayWeek: 1
  };
  public selectedDate: Date = this.getSelectedDate();
  public value07l: Date | null = new Date(this.selectedDate);

  private getSelectedDate(): Date {
    const d1: Date = new Date();
    const result: Date = new Date(d1.getFullYear(), d1.getMonth(), d1.getDate(), 0, 0, 0, 0);
    result.setDate(result.getDate() + (d1.getDate() === 1 ? 1 : -1));
    return result;
  }
}
