Like a Material-UI. #

It is possible to display GlnTextarea in style as "Material-UI".

This requires some css-styles to be defined.
exterior:   {{ labelOutlined }} {{ labelUnderline }} {{ labelStandard }}
State: empty.
Attributes: isRequired.
Color: primary or success.
Attribute: isDisabled.
{{ labelShowSource }} 
<form [formGroup]="formGroup03a"
  novalidate
  autocomplete="off">

  <div class="app-group ip-material-ui">

    <div class="app-elem ip-elem">
      <div>
        <small>
          <em>State: empty.</em>
        </small>
      </div>
      <gln-textarea formControlName="model03a"
        label="Enter value"
        helperText="Enter a value for the parameter."
        frameSize="wide"
        [exterior]="exterior03a"
        [minRows]="2"
        [maxRows]="4"
        wdFull>
      </gln-textarea>
    </div>

    <div class="app-elem ip-elem">
      <div>
        <small>
          <em>Attributes: isRequired.</em>
        </small>
      </div>
      <gln-textarea formControlName="model03b"
        isRequired
        [label]="!control03a.model03b.value
          ? 'Requires a value':'Parameter value'"
        [helperText]="control03a.model03b.value
          ? 'The field is filled.'
          : 'The field is not filled.'"
        frameSize="wide"
        [exterior]="exterior03a"
        [minRows]="2"
        [maxRows]="4"
        wdFull>
      </gln-textarea>
    </div>

    <div class="app-elem ip-elem">
      <div>
        <small>
          <em>Color: primary or success.</em>
        </small>
      </div>
      <gln-textarea formControlName="model03c"
        [ngClass]="[control03a['model03c'].value ? 'ip-success' : '']"
        [label]="!control03a.model03c.value
          ? 'Enter a parameter':'Parameter specified'"
        [helperText]="control03a.model03c.value
          ? 'The parameter was specified. (success)'
          : 'The parameter was not specified. (primary)'"
        frameSize="wide"
        [exterior]="exterior03a"
        [minRows]="2"
        [maxRows]="4"
        wdFull>
      </gln-textarea>
    </div>

    <div class="app-elem ip-elem">
      <div>
        <small>
          <em>Attribute: isDisabled.</em>
        </small>
      </div>
      <gln-textarea formControlName="model03d"
        label="Disabled"
        isDisabled
        helperText="The parameter is disabled."
        frameSize="wide"
        [exterior]="exterior03a"
        [minRows]="2"
        [maxRows]="4"
        wdFull>
      </gln-textarea>
    </div>

  </div>

</form>
 
import {
  Component, ViewEncapsulation
} from '@angular/core';
import {
  FormControl, FormGroup
} from '@angular/forms';

@Component({
  selector: 'app-textarea',
  templateUrl: './textarea.component.html',
  styleUrls: ['./textarea.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class TextareaComponent {
  public exterior03a = 'outlined';
  public control03a = {
    model03a: new FormControl(null, []),
    model03b: new FormControl(null, []),
    model03c: new FormControl('Hello World', []),
    model03d: new FormControl('Hello World', []),
    };
  public formGroup03a: FormGroup = new FormGroup(this.control03a);
}
 
.app-group {
  display: flex;
  flex-wrap: wrap;
}
.app-elem {
  min-width: 26ch;
  margin: 10px 10px;
}
.ip-elem {
  width: 29ch;
}
