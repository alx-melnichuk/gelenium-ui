Like a Bootstrap. #

It is possible to display GlnTextarea in style as "Bootstrap".
To do this, you need to specify the following configuration: 

public config02b: GlnFrameConfig = {
  exterior: GlnFrameExterior.outlined,
  frameSize: GlnFrameSize.short,
  isNoLabel: true,
};
 And also define some css-styles.
State: empty.
Attributes: isRequired.
Color: primary or success.
Attribute: isDisabled.
{{ labelShowSource }} 
<form [formGroup]="formGroup02a"
  novalidate
  autocomplete="off">

  <div class="app-group ip-bootstrap">

    <div class="app-elem ip-elem">
      <div>
        <small>
          <em>State: empty.</em>
        </small>
      </div>
      <gln-textarea formControlName="model02a"
        label="Enter value"
        helperText="Enter a value for the parameter."
        [config]="config02a"
        [minRows]="2"
        [maxRows]="4"
        wdFull>
      </gln-textarea>
    </div>

    <div class="app-elem ip-elem">
      <div>
        <small>
          <em>Attributes: isRequired.</em>
        </small>
      </div>
      <gln-textarea formControlName="model02b"
        isRequired
        [label]="!control02a.model02b.value
          ? 'Requires a value':'Parameter value'"
        [helperText]="control02a.model02b.value
          ? 'The field is filled.'
          : 'The field is not filled.'"
        [config]="config02a"
        [minRows]="2"
        [maxRows]="4"
        wdFull>
      </gln-textarea>
    </div>

    <div class="app-elem ip-elem">
      <div>
        <small>
          <em>Color: primary or success.</em>
        </small>
      </div>
      <gln-textarea formControlName="model02c"
        [ngClass]="[control02a['model02c'].value ? 'ip-success' : '']"
        [label]="!control02a.model02c.value
          ? 'Enter a parameter':'Parameter specified'"
        [helperText]="control02a.model02c.value
          ? 'The parameter was specified. (success)'
          : 'The parameter was not specified. (primary)'"
        [config]="config02a"
        [minRows]="2"
        [maxRows]="4"
        wdFull>
      </gln-textarea>
    </div>

    <div class="app-elem ip-elem">
      <div>
        <small>
          <em>Attribute: isDisabled.</em>
        </small>
      </div>
      <gln-textarea formControlName="model02d"
        label="Disabled"
        isDisabled
        helperText="The parameter is disabled."
        [config]="config02a"
        [minRows]="2"
        [maxRows]="4"
        wdFull>
      </gln-textarea>
    </div>

  </div>

</form>
 
import {
  Component, ViewEncapsulation
} from '@angular/core';
import {
  FormControl, FormGroup
} from '@angular/forms';
import {
  GlnFrameConfig, GlnFrameExterior, GlnFrameSize
} from 'gelenium-ui';

@Component({
  selector: 'app-textarea',
  templateUrl: './textarea.component.html',
  styleUrls: ['./textarea.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class TextareaComponent {
  public control02a = {
    model02a: new FormControl(null, []),
    model02b: new FormControl(null, []),
    model02c: new FormControl('Hello World', []),
    model02d: new FormControl('Hello World', []),
  };
  public formGroup02a: FormGroup = new FormGroup(this.control02a);
  public config02a: GlnFrameConfig = {
    exterior: GlnFrameExterior.outlined,
    frameSize: GlnFrameSize.short,
    isNoLabel: true,
  };

}
