import {
  Component, ViewEncapsulation
} from '@angular/core';
import { GlnSnackbarService } from 'gelenium-ui';

@Component({
  selector: 'app-snackbar',
  templateUrl: './snackbar.component.html',
  styleUrls: ['./snackbar.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class SnackbarComponent {

  constructor(private snackbarService: GlnSnackbarService) {}

  public clickBootstrap(
    message: string, action: string, config: any
  ): void {
    const configBS = {
      overlayClasses: 'snpl2-bs-overlay',
      horizontal: 'right',
      transition: 'fade',
      vertical: 'top',
    };
    this.snackbarService.open(message, action, { ...configBS, ...config });
  }
}
 
app-snackbar {
  .snpl2-elem {
    max-width: 40ch;
  }
}
